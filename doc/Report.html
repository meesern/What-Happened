def delta(later)
  t1_msb = self.known
  t1_lsb = self.second || 0.0
  t2_msb = later.known
  t2_lsb = later.second || 0.0

  d_msb = t2_msb - t1_msb
  d_lsb = t2_lsb - t1_lsb

  (d_msb + d_lsb)/1.0
end
