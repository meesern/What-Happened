def spanify(from,to)
  fr = Time.parse(from.andand.to_s || Time.at(0).to_s)
  tw = Time.parse(to.andand.to_s || "now")
  [fr,tw]
end

def time_iterate(start_time, end_time, step)
  idx = 0
  begin
    yield(start_time, idx)
    idx+=1
  end while (start_time += step) <= end_time
end
